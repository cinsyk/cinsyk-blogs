# 运算符

let number = 0;
number++; // 先使用后自增, result: 0;
console.log(number); // result:1
++number; // 先自增后使用, result: 2;
console.log(number); // 2
number--; // 先使用后自减, result: 2;
console.log(number); // 1
--number; // 先自减后使用, result: 0;
console.log(number); // result: 0
